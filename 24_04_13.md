```cpp
#include<iostream>
#include<vector>
#include<queue>
#include<cstring>
#include<algorithm>
#include<cmath>
using namespace std;

struct Info {
	int x, y;
	int dist;
	bool isEscape;

};

struct coord {
	int x, y;
};

coord Exit;
vector<Info>players;

int N, M, K, result;
int dx[]{ -1,1,0,0 };
int dy[]{ 0,0,-1,1 };

int Map[11][11];
int PMap[11][11];

void rotateclockwise(int x, int y, int len) {
	int tmp[11][11];

	for (int i = x; i < x + len; ++i) {
		for (int j = y; j < y + len; ++j) {
			int ti = i - x, tj = j - len;
			int oi = len - 1 - tj, oj = ti;
			tmp[i][j] = Map[oi + len][oj + len];
		}
	}

	for (int i = x; i < x + len; ++i) {
		for (int j = y; j < y + len; ++j) {
			Map[i][j] = tmp[i][j];
			if (Map[i][j] > 0)Map[i][j]--;
		}
	}


}

int main(void) {
	cin.tie(NULL)->sync_with_stdio(false);
	
	cin >> N >> M >> K;
	
	players.push_back({ -1,-1,0,true });
	
	for (int i = 1; i <= N; ++i) {
		for (int j = 1; j <= N; ++j) {
			cin >> Map[i][j];
		}
	}

	for (int i = 0; i < M; ++i) {
		int x, y;
		cin >> x >> y;
		players.push_back({ x,y,0,false });
		Map[x][y]++;
	}

	cin >> Exit.x >> Exit.y;
	Map[Exit.x][Exit.y] = 10;
	int count = 0;
	for (int k = 0; k < K; ++k) {//k턴 만큼 반복
		//탈출하지 않은 모든 참가자들 이동,거리+1
		for (auto& [x, y, dist, isEscape] : players) {
			if (isEscape)continue;
			auto& [ex, ey] = Exit;

			int curdist = abs(x - ex) + abs(y - ey);
			vector<int>dirs;
			for (int i = 0; i < 4; ++i) {
				int nx = x + dx[i], ny = y + dy[i];
				if (nx<1 || ny<1 || nx>N || ny>N)continue;
				if (Map[nx][ny] <= 9 && Map[nx][ny] >= 1)continue; // 벽이면 패스

				int mindist = abs(nx - ex) + abs(ny - ey);
				if (mindist < curdist)
					dirs.push_back(i);
			}

			if (dirs.size() == 0)continue;
			sort(dirs.begin(), dirs.end());
			int nx = x + dx[dirs.front()], ny = y + dy[dirs.front()];
			x = nx; y = ny;
			dist++;

			if (Map[nx][ny] == 10) {
				isEscape = true;
				count++;
			}

			if (count == M)
				break;

		}
		
		if (count == M)
			break;

		//정사각형 탐색 후 회전 및 내구도 줄이기
		for (int l = 2; l <= N; ++l) {
			bool checkE, checkP;

			for (int i = 1; i <= N; ++i) {
				for (int j = 1; j <= N; ++j) {//1,1부터 끝점까지 탐색,제일 좌상단에서 검사하는거라서 좌표 비교도 필요없음
					checkE = false; checkP = false;

					for (int rx = 0; rx < l; ++rx) {
						for (int ry = 0; ry < l; ++ry) {
							if (PMap[i + rx][j + ry])checkP = true;
							if (Map[i + rx][j + ry] == 10)checkE = true;
						}
					}

					if (checkE && checkP) {
						rotateclockwise(i, j, l);
						//출구랑 참가자도 같이 회전하니까 좌표 변경
					}
				}
			}
		}

	}

	for (auto& v : players) {
		result += v.dist;
	}

	cout << result << '\n';
	cout << Exit.x << Exit.y << '\n';

	return 0;
}
```

```cpp
int changedir(int dir) {
	if (dir == 0)
		return 2;
	else if (dir == 2)
		return 1;
	else if (dir == 1)
		return 3;
	else if (dir == 3)
		return 0;
}

void make_snail() {
	int x = N / 2;
	int y = N / 2;
	int movecnt = 1;
	int dir = 2;
	int Num = 0;

	snailMap[x][y] = Num;
	coords[Num] = { x,y };
	while (1) {
		for (int i = 0; i < 2; ++i) {
			for (int j = 0; j < movecnt; ++j) {
				x += dx[dir];
				y += dy[dir];
				snailMap[x][y] = ++Num;
				coords[Num] = { x,y };
			}
			dir = changedir(dir);
		}
		movecnt++;

		if (movecnt == N) {
			for (int j = 0; j < movecnt - 1; ++j) {
				x += dx[dir];
				y += dy[dir];
				snailMap[x][y] = ++Num;
				coords[Num] = { x,y };
			}
			break;
		}
	}
}

bool Breakballs() {
	bool canbreak = false;
	int curball = Map[coords[1].x][coords[1].y];
	int connection = 1;
	int start = 1;
	int Num;

	for (Num = 2; Num < N * N; ++Num) {
		int nball = Map[coords[Num].x][coords[Num].y];
		if (nball == 0)break;

		if (curball == nball)connection++;
		else {
			if (connection >= 4) {
				canbreak = true;
				for (int i = start; i < Num; ++i)Map[coords[i].x][coords[i].y] = -1;
				numcnt[curball] += connection;
			}

			curball = nball;
			connection = 1;
			start = Num;
		}

	}

	if (connection >= 4) {
		canbreak = true;
		for (int i = start; i < Num; ++i)Map[coords[i].x][coords[i].y] = -1;
		numcnt[curball] += connection;
	}

	return canbreak;
}

void renew_map() {
	int tmp_Map[60][60]{ 0, };
	int curball = Map[coords[1].x][coords[1].y];
	int count = 1;
	int Position = 1;
	bool check = true;

	for (int Num = 2; Num < N * N; ++Num) {
		if (Position >= N * N) {
			check = false;
			break;
		}

		int nball = Map[coords[Num].x][coords[Num].y];
		if (nball == 0)break;
		if (curball == nball)count++;
		else {
			tmp_Map[coords[Position].x][coords[Position].y] = count;
			tmp_Map[coords[Position + 1].x][coords[Position + 1].y] = curball;
			curball = nball;
			Position += 2;
		}
	}
	if (check == true) {
		if (Position != 1) {
			tmp_Map[coords[Position].x][coords[Position].y] = count;
			tmp_Map[coords[Position + 1].x][coords[Position + 1].y] = curball;
		}
	}

	for (int i = 0; i < N; ++i) {
		for (int j = 0; j < N; ++j) {
			Map[i][j] = tmp_Map[i][j];
		}
	}
}

		for (int t = 1; t <= s; ++t) {

		int cntminus = 0;
		bool check = false;

		//빈공간에 구슬땡겨오기(이게 관건)
		for (int i = 1; i < N * N; ++i) {
			auto& [cx, cy] = coords[i];
			auto& [nx, ny] = coords[i - cntminus];
			if (Map[cx][cy] == -1) {
				cntminus++;
				check = true;
			}
			else {
				if (Map[cx][cy] == 0) {
					//땡겨가서 빈 공간만큼 0으로 채워주기
					for (int j = 0; j < cntminus; ++j) Map[coords[i - j].x][coords[i - j].y] = 0;
					check = false;
					break;
				}
				else {
					Map[nx][ny] = Map[cx][cy];
				}
			}
		}
		//처음부터 배열이 꽉 차있던 경우
		if (check) {
			int i = (N * N) - 1;
			for (int j = 0; j < cntminus; j++, i--) Map[coords[i].x][coords[i].y] = 0;//뒤에서부터 부서진 부분만큼 채워주기
		}


		while (1) {
			//구슬 연속으로 있으면 터뜨리기
			if (Breakballs() == false)break;

			cntminus = 0;
			check = false;
			//빈공간에 구슬땡겨오기
			for (int i = 1; i < N * N; ++i) {
				auto& [cx, cy] = coords[i];
				auto& [nx, ny] = coords[i - cntminus];
				if (Map[cx][cy] == -1) {
					cntminus++;
					check = true;
				}
				else {
					if (Map[cx][cy] == 0) {
						//땡겨가서 빈 공간만큼 0으로 채워주기
						for (int j = 0; j < cntminus; ++j) Map[coords[i - j].x][coords[i - j].y] = 0;
						check = false;
						break;
					}
					else {
						Map[nx][ny] = Map[cx][cy];
					}
				}
			}
			//처음부터 배열이 꽉 차있던 경우
			if (check) {
				int i = (N * N) - 1;
				for (int j = 0; j < cntminus; j++, i--) Map[coords[i].x][coords[i].y] = 0;//뒤에서부터 부서진 부분만큼 채워주기
			}
		}

		//구슬 갯수 조사해서 정비하기
		renew_map();

	}

	int sum = numcnt[1] + 2 * numcnt[2] + 3 * numcnt[3];

	cout << sum << '\n';

	return 0;
```
